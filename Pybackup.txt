import os
import time


# PowerShell: Compress-Archive -Path F:\virustotal\ -DestinationPath F:\backip\backup\filename
# Linux zip: zip -qr

source = ["F:/virustotal/"]
target_dir = ["F:/backip/"]
target = "F:/backip/" + time.strftime("%Y%m%d") + ".zip"
print(str(target))
print(target_dir)
zip_cm = "zip -qr {0} {1}".format(target," ".join(source))
print(str(zip_cm))
if os.system(zip_cm) == 0:
    print("Good backup to: ", target)
else:
    print("Failed")
